✔ 5. Node.js 구현 구조
class AlsongProvider implements LyricsProvider {
  name = "alsong";
  priority = 3;

  async fetch(meta) {
    const xml = buildSoap(meta);
    const res = await axios.post(ALSONG_URL, xml);

    const lyric = parseXML(res.data);

    if (!lyric) return null;

    return {
      source: "alsong",
      lrc: lyric
    };
  }
}

import requests
from lxml import etree

ALSONG_URL = "http://lyrics.alsong.co.kr/alsongwebservice/service1.asmx"

SOAP_TEMPLATE = """<?xml version="1.0" encoding="UTF-8"?>
<SOAP-ENV:Envelope xmlns:SOAP-ENV="http://www.w3.org/2003/05/soap-envelope"
xmlns:ns1="ALSongWebServer">
<SOAP-ENV:Body>
<ns1:GetResembleLyric2>
<ns1:stQuery>
<ns1:strTitle>{title}</ns1:strTitle>
<ns1:strArtistName>{artist}</ns1:strArtistName>
<ns1:nCurPage>0</ns1:nCurPage>
</ns1:stQuery>
</ns1:GetResembleLyric2>
</SOAP-ENV:Body>
</SOAP-ENV:Envelope>
"""

class AlsongProvider(LyricsProvider):
    name = "alsong"

    def fetch(self, title, artist):
        try:
            res = requests.post(
                ALSONG_URL,
                data=SOAP_TEMPLATE.format(title=title, artist=artist),
                headers={"Content-Type": "application/soap+xml"},
                timeout=3
            )

            root = etree.fromstring(res.content)

            # XML에서 lyric 찾기
            lyric = root.xpath("//strLyric/text()")

            if not lyric:
                return None

            return lyric[0]

        except Exception:
            return None
